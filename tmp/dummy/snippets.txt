        private void btnLoad_Click(object sender, EventArgs e)
        {
            System.IO.DirectoryInfo ParentDirectory = new System.IO.DirectoryInfo(@"C:\Visual Studios\nuget\assemblies");

            foreach (System.IO.FileInfo f in ParentDirectory.GetFiles())
            {
                clbFiles.Items.Add(f.Name);
            }
        }

        private void btnNupecFile_Click(object sender, EventArgs e)
        {
            if(clbFiles.CheckedItems.Count != 0)
            {
                string s = "";
                for (int x = 0; x <= clbFiles.CheckedItems.Count - 1; x++)
                {
                    s = s + clbFiles.CheckedItems[x].ToString() + "\n";
                }
                laNuspecFile.Text = s;
            }
        }

        private void btnXmlReader_Click(object sender, EventArgs e)
        {
            OpenFileDialog ofd = new OpenFileDialog();
            ofd.Filter = "XML|*.xml";
            if (ofd.ShowDialog() == DialogResult.OK)
            {
                XmlDocument xDoc = new XmlDocument();
                xDoc.Load(ofd.FileName);
                MessageBox.Show(xDoc.SelectSingleNode("package/metadata/id").InnerText);
            }
        }
